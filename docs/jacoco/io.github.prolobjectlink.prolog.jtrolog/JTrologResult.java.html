package io.github.prolobjectlink.prolog.jtrolog;

import io.github.prolobjectlink.prolog.PrologProvider;
import io.github.prolobjectlink.prolog.PrologResult;
import io.github.prolobjectlink.prolog.PrologTerm;
import io.github.prolobjectlink.prolog.PrologTermType;
import io.github.prolobjectlink.prolog.PrologVariable;
import jTrolog.terms.Var;

public class JTrologResult extends JTrologVariable implements PrologResult {

	JTrologResult(PrologProvider provider, String name) {
		super(PrologTermType.RESULT_TYPE, provider, name, 0);
	}

	JTrologResult(PrologProvider provider, PrologTerm name) {
		super(PrologTermType.RESULT_TYPE, provider);
		this.value = new Var(((PrologVariable) name).getName(), 0);
	}

	JTrologResult(PrologProvider provider, int position) {
		super(PrologTermType.RESULT_TYPE, provider, "_", position);
	}

	JTrologResult(PrologProvider provider, String name, int position) {
		super(PrologTermType.RESULT_TYPE, provider, name, position);
	}

	public final PrologTerm getNameTerm() {
		return provider.newVariable(getName(), getPosition());
	}

}
